window.runSkill = async function(input) {
  try {
    const params = parseInput(input);
    const action = params.action;
    const date = resolveDate(params.date);
    const allData = getData();
    let resultMessage = '';
    let showDashboard = params.show_dashboard || action === 'get_history';
    if (action === 'log_mood') {
      const score = parseInt(params.score);
      if (isNaN(score) || score < 1 || score > 10) throw new Error('Mood score must be a number between 1 and 10.');
      allData[date] = { score, comment: params.comment || '', timestamp: new Date().toISOString() };
      saveData(allData);
      resultMessage = 'Logged mood of ' + score + '/10 for ' + date + (params.comment ? ': "' + params.comment + '"' : '') + '.';
    } else if (action === 'get_mood') {
      const entry = allData[date];
      resultMessage = entry ? ('Your mood on ' + date + ' was ' + entry.score + '/10' + (entry.comment ? ' (' + entry.comment + ')' : '') + '.') : ('No mood entry found for ' + date + '.');
    } else if (action === 'get_history') {
      const daysCount = parseInt(params.days) || 7;
      const sortedDates = Object.keys(allData).sort().slice(-daysCount);
      const history = sortedDates.map(d => ({ date: d, ...allData[d] }));
      resultMessage = 'Here is your mood history (oldest first, retrieved on ' + toLocalISO(new Date()) + '): \n' + history.map(h => '- ' + h.date + ': ' + h.score + '/10 ' + (h.comment ? '(' + h.comment + ')' : '')).join('\n');
    } else if (action === 'delete_mood') {
      if (allData[date]) { delete allData[date]; saveData(allData); resultMessage = 'Deleted mood entry for ' + date + '.'; }
      else resultMessage = 'No mood entry found to delete for ' + date + '.';
    } else if (action === 'export_data') {
      resultMessage = 'Here is your raw mood data for backup:\n\n' + JSON.stringify(allData, null, 2);
    } else if (action === 'wipe_data') {
      localStorage.removeItem(STORAGE_KEY);
      resultMessage = 'All your mood history has been deleted. You can now start fresh!';

      throw new Error('Unknown action: ' + action);
